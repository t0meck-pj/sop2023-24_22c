Uruchom program - nic nie podawaj na wejściu (program powinien czekać w nieskończoność).Uruchom nowy terminal.

#include <stdio.h>

int main(){

    int number;
    printf("PID: %d\n", getpid());
    printf("PPID: %d\n", getppid());
    scanf("%d", &number);
    

    return 0;
}
